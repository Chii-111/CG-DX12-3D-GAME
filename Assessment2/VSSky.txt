cbuffer SkyBuffer : register(b0)
{
    float4x4 WVP; 
};

struct VS_INPUT
{
    float3 Pos : POSITION;
    float2 Tex : TEXCOORD; 
};

struct PS_INPUT
{
    float4 Pos : SV_POSITION;
    float2 Tex : TEXCOORD;
};

PS_INPUT VS(VS_INPUT input)
{
    PS_INPUT output;
    
    output.Pos = mul(float4(input.Pos, 1.0f), WVP);
    
    output.Pos.z = output.Pos.w;
    
    output.Tex = input.Tex;
    
    return output;
}
